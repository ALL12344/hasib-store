function getImg(p){
if(p.cat===‘cpu’) return p.brand===‘amd’?S.cpu_amd:S.cpu_intel;
if(p.cat===‘gpu’){
if(p.brand===‘amd’) return S.gpu_amd;
if(p.nameEn.includes(‘GTX’)) return S.gpu_gtx;
return S.gpu_nvidia;
}
if(p.cat===‘ram’) return S.ram;
if(p.cat===‘storage’){
if(p.nameEn.includes(‘HDD’)) return S.hdd;
if(p.nameEn.includes(‘PCIe 5’)) return S.nvme_fast;
if(p.nameEn.includes(‘NVMe’)||p.nameEn.includes(‘M.2’)) return S.ssd_nvme;
return S.ssd_sata;
}
if(p.cat===‘mb’) return S.mb;
if(p.cat===‘cooling’) return p.name.includes(‘AIO’)||p.name.includes(‘360’)||p.name.includes(‘240’)?S.cooling_aio:S.cooling_air;
if(p.cat===‘psu’) return S.psu;
if(p.cat===‘case’) return S.case;
return S.cpu_intel;
}
